## Pure arch linux installation guide  

# ping -c 4 google.com
# timedatectl set-ntp true
# timedatectl status
# cfdisk (manage disk partitions)
# mkfs.ext4 /dev/sda1
# mkswap /dev/sda2 
# swapon /dev/sda2
# mount /dev/sda1 /mnt
# pacstrap /mnt base linux linux-firmware
# genfstab -U /mnt >> /mnt/etc/fstab
# arch-chroot /mnt
# ln -sf /usr/share/zoneinfo/Region/City /etc/localtime
# hwclock --systohc

# nano /etc/locale.gen
    (uncomment en_US.UTF-8 UTF8)

# locale-gen

# nano /etc/locale.conf
LANG=en_US.UTF-8

# nano /etc/hostname
myhostname  (rise was what I used)

# nano /etc/hosts
127.0.0.1 localhost
::1          localhost
127.0.1.1 myhostname.localdomain myhostname

# passwd
# useradd -m newusername (I used rise)
# passwd yourusername
# usermod -aG wheel,audio,video,optical,storage yourusername
# pacman -S sudo
# pacman -S vim
# visudo (uncomment %wheel All=(All) All
                    :wq)
# pacman -S ntfs-3g
# pacman -S grub
# grub-install /dev/sda
# grub-mkconfig -o /boot/grub/grub.cfg
# pacman -S os-prober
# os-prober
# grub-mkconfig -o /boot/grub/grub.cfg
# pacman -S xorg 
# exit
# reboot
